В ini_voice.xml

TopMost = "true" - отображать окно поверх остальных окон
	= "false" - стандартный порядок отображения окна

ShowOnCall = "true" - разворачивать при вызове
	   = "false" - не разворачивать при вызове, разворачивать только при клике на иконку в трее

LogPath - путь для сохранения записей
